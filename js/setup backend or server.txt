// STEPS FOR BACKEND SETUP
1st step: Create a folder named backend or server
2nd step: We should go inside backend dir / `cd backend`
3rd step: package initialize `npm init`
4th step: Install express `npm i express`
5th step: Create a file named index.js in backend dir
6th step: We will work in ES Module - import , export so, we have to go inside package.json and update "type":"commonjs" to  "type":"module"
7th step (Server Setup) : We will create a variable app and give it express functionality
`const app = express()` // Do not forget to import express
8th step: Run express file in specific port, 
`
const PORT = 5000 //PORT bhanney variable lai 5000 assign garey

app.listen(PORT, () => { // 5000 port number aaucha ani backend le tellai listen garera 5000 port ma run garcha
    console.log(`App is runing on port: ${PORT}`) // PORT aayo bhani rw backend le PORT listen garyo bhani console ma message print gardencha
} )
